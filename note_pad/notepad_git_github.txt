git + github 활용법

# git을 통해 정리하고 싶은 '디렉토리'에서 "git bash"을 선택해 실행 시작할 것

------------------------------------------------------------------------
git bash실행

git init

git config --global user.email "ho8566z@gmail.com"

git config --global user.name "ho8566z"

git config --list
#email, name 확인

------------------------------------------------------------------------
git status

git add .
#각각 추가할 때: git add (ex: doc.txt)

git commit -m "(ex)20260508 task complete"

------------------------------------------------------------------------
git log
#git commit 확인

github에 'sign in'하고 'repositories'개설할 것
#chrome에서 확인

git remote add origin https://~(github: repositories 코드)

git remote -v
#fetch, push 확인

git branch -M main
#master -> main 변경

git push origin main
#github에 업로드

------------------------------------------------------------------------
git clone https://~(github: repositories 코드) .
#github에서 '처음' 다운로드
#마지막에 "." 찍기

git pull origin main
#github에서 '다시' 다운로드

------------------------------------------------------------------------
------------------------------------------------------------------------
git checkout (특정commit번호)
#특정 commit으로 이동

git reflog
#사라진 log기록 전체 확인

git checkout master
#head -> master 관계 재설정

git reset --hard (특정commit번호)
#특정 commit으로 이동(관계 유지)

------------------------------------------------------------------------
git restore --staged doc.txt
#특정 파일, 디렉토리 '스테이지(add상태)'에서 add상태 이전으로

------------------------------------------------------------------------
------------------------------------------------------------------------
gitignore
#git 디렉토리에서 github에 업로드를 원치 않는 파일, 디렉토리 업로드를 막는 것
#새로 txt를 만들고 파일명을 ".gitignore"로 설정하고,
#새 txt에 '.gitignore'를 드래그 해서 실행하고, 원치 않는 파일명, 디렉토리명을 
#입력하면 완료
#-> git status, add, commit 진행하고, push진행하고, 
#repositories에서 진짜 없는지 확인할 것

MS VScode에서의 git bash : add + commit 동시에 하는법

PS C:\lyh\python\python_ex> git commit -am "20260521 14:49"
--> git commit -am "~"

PS C:\lyh\python\python_ex> git add .; git commit -m "20260521 14:50"
--> git add .; git commit -m "~"
